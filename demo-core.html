<script>

const $ = s => document.querySelector(s);

$('#chain').value = 'demo-core-' + new Date().toISOString().replace(/[-:TZ.]/g,'').slice(0,12);


function add(msg){ $('#log').textContent += '\n' + msg; }


$('#send').onclick = async ()=>{

  const chain = $('#chain').value.trim();

  $('#log').textContent = `Creating receipts for chain: ${chain}`;

  const now = new Date().toISOString();


  const demoEvents = [

    {module:"core", event:"decision_inferred", actor_id:"svc-recommender", subject_id:"user-123", reason:"initial scoring", score:0.81, timestamp:now},

    {module:"core", event:"human_review", reviewer_id:"ops-7", subject_id:"user-123", action:"approve", note:"meets policy", timestamp:now},

    {module:"core", event:"decision_finalized", subject_id:"user-123", outcome:"approved", timestamp:now}

  ];


  // Simulate success (no server needed)

  let prev = null;

  for (let i=0;i<demoEvents.length;i++){

    const e = demoEvents[i];

    // fake hash linking

    const fakeHash = btoa(JSON.stringify(e)).slice(0,16);

    const link = prev ? ' ←prev ' + prev : '';

    add(`✅ [${i+1}/3] ${e.event}  (hash ${fakeHash}${link})`);

    prev = fakeHash;

    await new Promise(r=>setTimeout(r,350));

  }

  add('\n' + 'All events appended. Now click “Export (download sample ZIP)”.');

};


add('Tip: After download, open the ZIP → open verifier.html → load the chain file to see integrity OK.');

</script>
